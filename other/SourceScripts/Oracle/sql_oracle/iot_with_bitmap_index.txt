  START WITH 1
  MAXVALUE 9999999999999999999999999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;

(

  EMPLOYEE_NO  NUMBER (8),

  CONSTRAINT     PK_SIMPLE_IOT PRIMARY KEY (EMPLOYEE_NO)
